-- Query 1: Top locations with call counts and caller information
SELECT ROUND(CAST(c.LAT AS DOUBLE), 3) AS latitude, ROUND(CAST(c.LON AS DOUBLE), 3) AS longitude, COUNT(*) AS call_count, STRING_AGG(DISTINCT c.callingNumber, ',') AS unique_callers, COUNT(DISTINCT c.callingNumber) AS unique_caller_count, MIN(c.eventTime) AS first_call, MAX(c.eventTime) AS last_call FROM cdr c JOIN connections_with_confidence cwc ON c.callingNumber = cwc.phone_number GROUP BY latitude, longitude ORDER BY call_count DESC LIMIT 10;

-- Query 2: Mobile devices with multiple unique locations
SELECT cwc.maid_id, cwc.phone_number, COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) AS unique_locations, COUNT(*) AS total_calls, MIN(c.eventTime) AS first_call, MAX(c.eventTime) AS last_call FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id, cwc.phone_number HAVING unique_locations > 1 ORDER BY unique_locations DESC, total_calls DESC LIMIT 20;

-- Query 3: Time difference analysis between MAID and CDR records
SELECT cwc.maid_id, cwc.phone_number, cwc.maid_timestamp, cwc.cdr_timestamp, DATE_DIFF('day', CAST(REPLACE(cwc.cdr_timestamp, 'T', ' ') AS TIMESTAMP), CAST(cwc.maid_timestamp AS TIMESTAMP)) AS days_difference, DATE_DIFF('hour', CAST(REPLACE(cwc.cdr_timestamp, 'T', ' ') AS TIMESTAMP), CAST(cwc.maid_timestamp AS TIMESTAMP)) AS hours_difference, CAST(m.LATITUDE AS DOUBLE) AS maid_latitude, CAST(m.LONGITUDE AS DOUBLE) AS maid_longitude, CAST(c.LAT AS DOUBLE) AS call_latitude, CAST(c.LON AS DOUBLE) AS call_longitude, cwc.distance AS location_difference FROM connections_with_confidence cwc JOIN maid m ON cwc.maid_id = m.ID JOIN cdr c ON cwc.phone_number = c.callingNumber WHERE cwc.maid_timestamp IS NOT NULL AND cwc.cdr_timestamp IS NOT NULL ORDER BY ABS(days_difference) LIMIT 20;

-- Query 4: Device statistics by make/model/OS
SELECT m.DEVICE_MAKE, m.DEVICE_MODEL, m.DEVICE_OS, COUNT(DISTINCT cwc.maid_id) AS unique_devices, COUNT(DISTINCT cwc.phone_number) AS unique_phone_numbers, COUNT(*) AS total_mappings, ROUND(AVG(CAST(m.LATITUDE AS DOUBLE)), 3) AS avg_latitude, ROUND(AVG(CAST(m.LONGITUDE AS DOUBLE)), 3) AS avg_longitude FROM connections_with_confidence cwc JOIN maid m ON cwc.maid_id = m.ID WHERE m.DEVICE_MAKE IS NOT NULL GROUP BY m.DEVICE_MAKE, m.DEVICE_MODEL, m.DEVICE_OS ORDER BY unique_devices DESC, total_mappings DESC LIMIT 15;

-- Query 5: Hourly call patterns
SELECT SUBSTRING(REPLACE(c.eventTime, 'T', ' '), 12, 2) AS hour_of_day, COUNT(*) AS call_count, COUNT(DISTINCT c.callingNumber) AS unique_callers, COUNT(DISTINCT cwc.maid_id) AS unique_devices FROM cdr c JOIN connections_with_confidence cwc ON c.callingNumber = cwc.phone_number GROUP BY hour_of_day ORDER BY hour_of_day;

-- Query 6: Mobile MAID devices with multiple locations and phone associations
SELECT m.ID AS maid_id, COUNT(DISTINCT ROUND(CAST(m.LATITUDE AS DOUBLE), 3) || ',' || ROUND(CAST(m.LONGITUDE AS DOUBLE), 3)) AS unique_locations, MIN(m."TIMESTAMP") AS first_seen, MAX(m."TIMESTAMP") AS last_seen, COUNT(DISTINCT cwc.phone_number) AS associated_phone_numbers, STRING_AGG(DISTINCT cwc.phone_number, ',') AS phone_numbers FROM maid m JOIN connections_with_confidence cwc ON m.ID = cwc.maid_id GROUP BY m.ID HAVING unique_locations > 1 AND associated_phone_numbers > 0 ORDER BY unique_locations DESC, associated_phone_numbers DESC LIMIT 15;

-- Query 7: MAID devices associated with multiple phone numbers
SELECT cwc.maid_id, COUNT(DISTINCT cwc.phone_number) AS unique_phone_numbers, STRING_AGG(DISTINCT cwc.phone_number, ',') AS phone_numbers, ROUND(AVG(CAST(m.LATITUDE AS DOUBLE)), 4) AS avg_latitude, ROUND(AVG(CAST(m.LONGITUDE AS DOUBLE)), 4) AS avg_longitude, ANY_VALUE(m.DEVICE_MAKE) AS device_make, ANY_VALUE(m.DEVICE_MODEL) AS device_model FROM connections_with_confidence cwc JOIN maid m ON cwc.maid_id = m.ID GROUP BY cwc.maid_id HAVING unique_phone_numbers > 1 ORDER BY unique_phone_numbers DESC LIMIT 15;

-- Query 8: Device OS analysis with location accuracy
SELECT m.DEVICE_OS, COUNT(DISTINCT cwc.maid_id) AS unique_devices, COUNT(DISTINCT cwc.phone_number) AS unique_phone_numbers, COUNT(*) AS total_mappings, ROUND(AVG(cwc.distance), 4) AS avg_location_difference FROM connections_with_confidence cwc JOIN maid m ON cwc.maid_id = m.ID WHERE m.DEVICE_OS IS NOT NULL GROUP BY m.DEVICE_OS ORDER BY unique_devices DESC;

-- Query 9: Highly mobile phone numbers (50+ unique locations per day)
SELECT c.callingNumber, SUBSTRING(c.eventTime, 1, 10) AS call_date, COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) AS unique_locations, MIN(c.eventTime) AS first_call, MAX(c.eventTime) AS last_call, COUNT(*) AS total_calls FROM cdr c GROUP BY c.callingNumber, call_date HAVING unique_locations >= 50 ORDER BY unique_locations DESC, total_calls DESC LIMIT 100;

-- Query 10: Highly mobile phone numbers and their connected MAID devices
WITH mobile_numbers AS (SELECT c.callingNumber, COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) AS unique_locations FROM cdr c GROUP BY c.callingNumber HAVING unique_locations >= 50) SELECT mn.callingNumber, ANY_VALUE(mn.unique_locations) AS unique_locations, COUNT(DISTINCT cwc.maid_id) AS connected_maids, STRING_AGG(DISTINCT cwc.maid_id, ',') AS maid_ids, COUNT(*) AS total_connections FROM mobile_numbers mn JOIN connections_with_confidence cwc ON mn.callingNumber = cwc.phone_number GROUP BY mn.callingNumber ORDER BY ANY_VALUE(mn.unique_locations) DESC, connected_maids DESC LIMIT 50;

-- Query 11: Top mobile numbers with their MAID location patterns
WITH mobile_numbers AS (SELECT c.callingNumber, COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) AS unique_locations FROM cdr c GROUP BY c.callingNumber HAVING unique_locations >= 50 LIMIT 10) SELECT mn.callingNumber, ANY_VALUE(mn.unique_locations) AS cdr_locations, cwc.maid_id, COUNT(DISTINCT ROUND(CAST(m.LATITUDE AS DOUBLE), 3) || ',' || ROUND(CAST(m.LONGITUDE AS DOUBLE), 3)) AS maid_locations, MIN(m."TIMESTAMP") AS first_seen, MAX(m."TIMESTAMP") AS last_seen FROM mobile_numbers mn JOIN connections_with_confidence cwc ON mn.callingNumber = cwc.phone_number JOIN maid m ON cwc.maid_id = m.ID GROUP BY mn.callingNumber, cwc.maid_id ORDER BY ANY_VALUE(mn.unique_locations) DESC, maid_locations DESC LIMIT 100;

-- Query 12: Starting locations of highly mobile devices
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50) SELECT ROUND(CAST(m.LATITUDE AS DOUBLE), 3) AS start_latitude, ROUND(CAST(m.LONGITUDE AS DOUBLE), 3) AS start_longitude, COUNT(DISTINCT md.maid_id) AS device_count, STRING_AGG(DISTINCT md.maid_id, ',') AS device_ids FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID JOIN (SELECT ID AS maid_id, SUBSTRING(m2."TIMESTAMP", 1, 10) AS day, MIN(m2."TIMESTAMP") AS first_timestamp FROM maid m2 GROUP BY maid_id, day) first_loc ON m.ID = first_loc.maid_id AND m."TIMESTAMP" = first_loc.first_timestamp GROUP BY start_latitude, start_longitude HAVING device_count > 1 ORDER BY device_count DESC LIMIT 50;

-- Query 13: Ending locations of highly mobile devices
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50) SELECT ROUND(CAST(m.LATITUDE AS DOUBLE), 3) AS end_latitude, ROUND(CAST(m.LONGITUDE AS DOUBLE), 3) AS end_longitude, COUNT(DISTINCT md.maid_id) AS device_count, STRING_AGG(DISTINCT md.maid_id, ',') AS device_ids FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID JOIN (SELECT ID AS maid_id, SUBSTRING(m2."TIMESTAMP", 1, 10) AS day, MAX(m2."TIMESTAMP") AS last_timestamp FROM maid m2 GROUP BY maid_id, day) last_loc ON m.ID = last_loc.maid_id AND m."TIMESTAMP" = last_loc.last_timestamp GROUP BY end_latitude, end_longitude HAVING device_count > 1 ORDER BY device_count DESC LIMIT 50;

-- Query 14: Device characteristics of highly mobile devices
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50) SELECT m.DEVICE_MAKE, m.DEVICE_MODEL, m.DEVICE_OS, COUNT(DISTINCT md.maid_id) AS device_count FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID WHERE m.DEVICE_MAKE IS NOT NULL GROUP BY m.DEVICE_MAKE, m.DEVICE_MODEL, m.DEVICE_OS ORDER BY device_count DESC LIMIT 50;

-- Query 15: Popular locations among highly mobile devices
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50) SELECT ROUND(CAST(m.LATITUDE AS DOUBLE), 3) AS latitude, ROUND(CAST(m.LONGITUDE AS DOUBLE), 3) AS longitude, COUNT(DISTINCT md.maid_id) AS unique_devices, COUNT(*) AS total_visits FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID GROUP BY latitude, longitude HAVING unique_devices > 1 ORDER BY total_visits DESC, unique_devices DESC LIMIT 50;

-- Query 16: Movement transitions between locations for mobile devices
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50 LIMIT 5), device_locations AS (SELECT md.maid_id, m."TIMESTAMP", ROUND(CAST(m.LATITUDE AS DOUBLE), 3) AS latitude, ROUND(CAST(m.LONGITUDE AS DOUBLE), 3) AS longitude, ROW_NUMBER() OVER (PARTITION BY md.maid_id ORDER BY m."TIMESTAMP") AS location_seq FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID) SELECT dl1.maid_id, dl1.latitude AS from_latitude, dl1.longitude AS from_longitude, dl2.latitude AS to_latitude, dl2.longitude AS to_longitude, COUNT(*) AS transition_count FROM device_locations dl1 JOIN device_locations dl2 ON dl1.maid_id = dl2.maid_id AND dl1.location_seq + 1 = dl2.location_seq GROUP BY dl1.maid_id, dl1.latitude, dl1.longitude, dl2.latitude, dl2.longitude HAVING transition_count > 1 ORDER BY transition_count DESC LIMIT 100;

-- Query 17: Time spent at each location for mobile devices
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50 LIMIT 10), location_times AS (SELECT md.maid_id, ROUND(CAST(m.LATITUDE AS DOUBLE), 3) AS latitude, ROUND(CAST(m.LONGITUDE AS DOUBLE), 3) AS longitude, MIN(m."TIMESTAMP") AS first_seen, MAX(m."TIMESTAMP") AS last_seen, DATE_DIFF('minute', CAST(MIN(m."TIMESTAMP") AS TIMESTAMP), CAST(MAX(m."TIMESTAMP") AS TIMESTAMP)) AS minutes_at_location FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID GROUP BY md.maid_id, latitude, longitude) SELECT maid_id, latitude, longitude, first_seen, last_seen, ROUND(minutes_at_location, 2) AS minutes_at_location FROM location_times WHERE minutes_at_location > 0 ORDER BY maid_id, minutes_at_location DESC LIMIT 100;

-- Query 18: Weekday vs weekend mobility patterns
WITH mobile_devices AS (SELECT cwc.maid_id FROM connections_with_confidence cwc JOIN cdr c ON cwc.phone_number = c.callingNumber GROUP BY cwc.maid_id HAVING COUNT(DISTINCT ROUND(CAST(c.LAT AS DOUBLE), 3) || ',' || ROUND(CAST(c.LON AS DOUBLE), 3)) >= 50), device_day_locations AS (SELECT md.maid_id, SUBSTRING(m."TIMESTAMP", 1, 10) AS day, CASE WHEN DAYOFWEEK(CAST(SUBSTRING(m."TIMESTAMP", 1, 10) AS DATE)) IN (1, 7) THEN 'weekend' ELSE 'weekday' END AS day_type, COUNT(DISTINCT ROUND(CAST(m.LATITUDE AS DOUBLE), 3) || ',' || ROUND(CAST(m.LONGITUDE AS DOUBLE), 3)) AS unique_locations FROM mobile_devices md JOIN maid m ON md.maid_id = m.ID GROUP BY md.maid_id, day) SELECT maid_id, day_type, AVG(unique_locations) AS avg_locations_per_day, MAX(unique_locations) AS max_locations_per_day, MIN(unique_locations) AS min_locations_per_day, COUNT(DISTINCT day) AS days_counted FROM device_day_locations GROUP BY maid_id, day_type HAVING days_counted > 0 ORDER BY maid_id, day_type LIMIT 100;
